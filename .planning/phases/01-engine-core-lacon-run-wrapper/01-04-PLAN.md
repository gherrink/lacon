```rust
//! Starlark `post_process` host. Hermetic mode: Globals::standard() only.
//!
//! Per CONTEXT.md D-08 + ADR-0003 + ADR-0008:
//! - The Starlark function operates on the aggregated post-pipeline output.
//! - No per-line evaluation (out of v1 scope).
//! - No `load`, no file I/O, no `print` — `Globals::standard()` provides
//!   none of these, and we deliberately do NOT call `eval.set_loader()`.
//!
//! Cold-start cost is measured in PLAN-07 benchmark task; if >2ms, the
//! StarlarkScript is constructed lazily per-invocation (not at rule load).
//! Currently the AstModule is parsed once at rule load (D-14) and the VM
//! is constructed fresh per `run()` call.

use starlark::environment::{Globals, Module};
use starlark::eval::Evaluator;
use starlark::syntax::{AstModule, Dialect};
use std::path::PathBuf;

use crate::error::{RuntimeError, ValidationError};

pub struct StarlarkScript {
    ast: AstModule,
    function_name: String,
    source_path: PathBuf,
}

#[derive(Debug, Clone)]
pub struct ScriptCtx {
    pub exit_code: i32,
    pub duration_ms: u64,
    pub command: String,
    pub args: Vec<String>,
    pub project_path: Option<String>,
}

impl Default for ScriptCtx {
    fn default() -> Self {
        Self {
            exit_code: 0, duration_ms: 0,
            command: String::new(), args: Vec::new(),
            project_path: None,
        }
    }
}

impl StarlarkScript {
    pub fn parse(
        content: &str,
        function_name: String,
        source_path: PathBuf,
    ) -> Result<Self, ValidationError> {
        // Pitfall 4: AstModule::parse takes owned String.
        let display_name = source_path.display().to_string();
        match AstModule::parse(&display_name, content.to_owned(), &Dialect::Standard) {
            Ok(ast) => Ok(Self { ast, function_name, source_path }),
            Err(e) => Err(ValidationError::ParseError {
                path: source_path,
                line: 0, // starlark::Error doesn't expose line/col directly in 0.13 stable API
                message: e.to_string(),
            }),
        }
    }

    pub fn run(
        &self,
        ctx: &ScriptCtx,
        lines: Vec<String>,
    ) -> Result<Vec<String>, RuntimeError> {
        let globals = Globals::standard();
        let module = Module::new();

        // Evaluate the module body (defines `process` etc.) into the module env.
        {
            let mut eval = Evaluator::new(&module);
            // The AstModule must be owned by the evaluator call; clone the AST.
            // starlark 0.13 supports cheap clone (AstModule is Arc-backed internally).
            let ast = self.ast.clone();
            eval.eval_module(ast, &globals).map_err(|e| RuntimeError::StarlarkEvalError {
                path: self.source_path.clone(),
                message: e.to_string(),
            })?;
        }

        // Look up `process` in the module's namespace.
        let process_val = module.get(&self.function_name).ok_or_else(|| {
            RuntimeError::StarlarkEvalError {
                path: self.source_path.clone(),
                message: format!("function `{}` not defined", self.function_name),
            }
        })?;

        // Build ctx as a Starlark dict.
        let heap = module.heap();
        let ctx_dict = {
            use starlark::values::dict::DictRef;
            // Construct via small-dict pattern: build (key, value) pairs and
            // call the heap's dict allocation. Exact API: heap.alloc(SmallMap{...}).
            // For starlark 0.13 use `Dict::new(SmallMap::new())` or similar; the
            // simplest cross-version path is to evaluate a Starlark expression
            // that builds the dict and pass that value. Use eval_module pattern:
            // pre-set named values on the module via module.set, then read.
            //
            // Simpler approach for v1: set ctx fields as separate names.
            //   ctx_exit_code, ctx_duration_ms, ctx_command, ctx_args, ctx_project_path
            // But the spec calls for `ctx.exit_code`. Fall back to a dict literal
            // built at runtime via eval of a tiny synthetic source:

            let tiny_src = format!(
                r#"_ctx = dict(exit_code = {}, duration_ms = {}, command = "{}", args = [{}], project_path = {})"#,
                ctx.exit_code,
                ctx.duration_ms,
                escape_starlark_string(&ctx.command),
                ctx.args.iter()
                    .map(|s| format!(r#""{}""#, escape_starlark_string(s)))
                    .collect::<Vec<_>>()
                    .join(", "),
                match &ctx.project_path {
                    Some(p) => format!(r#""{}""#, escape_starlark_string(p)),
                    None => "None".to_owned(),
                }
            );
            let ctx_ast = AstModule::parse("_ctx", tiny_src, &Dialect::Standard)
                .map_err(|e| RuntimeError::StarlarkEvalError {
                    path: self.source_path.clone(),
                    message: format!("ctx assembly failed: {}", e),
                })?;
            let mut eval = Evaluator::new(&module);
            eval.eval_module(ctx_ast, &globals).map_err(|e| RuntimeError::StarlarkEvalError {
                path: self.source_path.clone(),
                message: format!("ctx eval failed: {}", e),
            })?;
            module.get("_ctx").ok_or_else(|| RuntimeError::StarlarkEvalError {
                path: self.source_path.clone(),
                message: "ctx not bound".to_owned(),
            })?
        };

        // Allocate lines as a Starlark list.
        let lines_val = {
            let allocated: Vec<_> = lines.iter().map(|s| heap.alloc(s.as_str())).collect();
            heap.alloc_list(&allocated)
        };

        // Call process(ctx, lines).
        let result_val = {
            let mut eval = Evaluator::new(&module);
            eval.eval_function(process_val, &[ctx_dict, lines_val], &[])
                .map_err(|e| RuntimeError::StarlarkEvalError {
                    path: self.source_path.clone(),
                    message: e.to_string(),
                })?
        };

        // Extract result as Vec<String>.
        let result_iter = result_val.iterate(heap).map_err(|e| RuntimeError::StarlarkResultTypeError {
            path: self.source_path.clone(),
            function: self.function_name.clone(),
            got: format!("{} (iterate failed: {})", result_val.get_type(), e),
        })?;

        let mut out = Vec::new();
        for v in result_iter {
            let s = v.unpack_str().ok_or_else(|| RuntimeError::StarlarkResultTypeError {
                path: self.source_path.clone(),
                function: self.function_name.clone(),
                got: format!("list element of type {}", v.get_type()),
            })?;
            out.push(s.to_owned());
        }
        Ok(out)
    }
}

fn escape_starlark_string(s: &str) -> String {
    s.replace('\\', "\\\\").replace('"', "\\\"").replace('\n', "\\n")
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn identity_round_trip() {
        let script = StarlarkScript::parse(

        ).unwrap();

        assert!(matches!(err, RuntimeError::StarlarkResultTypeError { .. }));
    }

    #[test]

        // `load` is forbidden in Standard dialect when no loader is set —
        // verifies T-02 mitigation: hermetic by construction.
        let result = StarlarkScript::parse(
            "load(\"foo.bzl\", \"bar\")\ndef process(ctx, lines):\n    return lines\n",
            "process".to_owned(),
            std::path::PathBuf::from("loady.star"),
        );
        // Either parse rejects (Standard dialect disallows load when no loader), or
        // eval rejects (no loader configured). Either branch satisfies the threat
        // mitigation. Accept both:
        match result {
            Err(_) => {} // parse-time rejection — fine
            Ok(script) => {
                let err = script.run(&ScriptCtx::default(), vec![]).unwrap_err();
                assert!(matches!(err, RuntimeError::StarlarkEvalError { .. }));
            }
        }
    }
}
```

```rust
    /// Run lines through native stages, then optionally invoke a Starlark
    /// post_process script on the aggregated output.
    ///
    /// Per CON-filter-rule-post-process, post_process runs ONCE on the
    /// aggregated output. Per CON-filter-rule-native-primitives, max_bytes
    /// should be the LAST stage; this means post_process can re-grow
    /// output above max_bytes — that is documented behaviour. If a rule
    /// author wants the cap enforced after Starlark, they should declare
    /// `max_bytes` in `post_process` is impossible (the YAML schema places
    /// post_process outside the pipeline). If this becomes a real concern,
    /// re-cap after Starlark; for v1 we follow the spec literally.
    pub fn run_with_post_process(
        &mut self,
        lines: impl Iterator<Item = String>,
        post_process: Option<&crate::starlark_host::StarlarkScript>,
        ctx: &crate::starlark_host::ScriptCtx,
    ) -> Result<Vec<String>, crate::error::RuntimeError> {
        let aggregated = self.run(lines);
        match post_process {
            Some(script) => script.run(ctx, aggregated),
            None => Ok(aggregated),
        }
    }
}
```

```rust
fn resolve_script(
    spec: &schema::ScriptSpec,
    rule_source_path: &Path,
) -> Result<crate::starlark_host::StarlarkScript, ValidationError> {
    if spec.path.is_absolute() {
        return Err(ValidationError::MissingScriptFile {
            path: rule_source_path.to_owned(),
            line: 0,
            message: format!("script path must be relative: {}", spec.path.display()),
        });
    }
    if spec.path.components().any(|c| matches!(c, std::path::Component::ParentDir)) {
        return Err(ValidationError::MissingScriptFile {
            path: rule_source_path.to_owned(),
            line: 0,
            message: format!("script path must not contain `..`: {}", spec.path.display()),
        });
    }
    let resolved = rule_source_path
        .parent()
        .ok_or_else(|| ValidationError::MissingScriptFile {
            path: rule_source_path.to_owned(),
            line: 0,
            message: "rule file has no parent directory".to_owned(),
        })?
        .join(&spec.path);
    let content = std::fs::read_to_string(&resolved).map_err(|_| {
        ValidationError::MissingScriptFile {
            path: resolved.clone(),
            line: 0,
            message: format!("script file not found: {}", resolved.display()),
        }
    })?;
    crate::starlark_host::StarlarkScript::parse(&content, spec.function.clone(), resolved)
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User-authored .star scripts | Referenced by `script:` / `post_process:` in rule YAML; loaded from disk relative to the rule file. The rule author trusts their own scripts; the engine treats scripts as untrusted code that must execute hermetically. |
| Starlark VM execution | A bounded sandbox: no I/O, no network, no `load`, no `print`. Exhaustion of CPU/memory is the only attack surface. |

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-01 | Information Disclosure | Untrusted .star reading filesystem | mitigate | `Globals::standard()` does not include file I/O. Code does NOT call `eval.set_loader()`. Test `load_statement_rejected_by_hermetic_runtime` proves the runtime rejects `load`. Negative grep gate (`grep -c 'set_loader'` returns 0) prevents regression. |
| T-04-02 | Denial of Service | Untrusted .star with infinite loop | accept (v1) | starlark-rust 0.13 does not expose a stable instruction-count cap in v1 of this plan. The `RuntimeError::StarlarkResourceLimit` variant exists for forward-compat; PLAN-07 benchmark task measures observed cold-start of trivial `process()`. If a rule author writes an infinite loop, `lacon run` hangs — same as if they wrote a misbehaving native command. Documented in `docs/architecture.md` post-implementation. |
| T-04-03 | Path Traversal | `script:` / `post_process: { path: ... }` field | mitigate | `resolve_script` rejects absolute paths and any path containing `..`. Reuses the same path-traversal guard as PLAN-03 task 2 for consistency. |
