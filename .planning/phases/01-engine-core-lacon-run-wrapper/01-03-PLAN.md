pub struct RuleFile {

    #[serde(default)] pub pipeline: Option<Vec<StageSpec>>,

}

#[derive(Deserialize, Debug, Clone)]
#[serde(deny_unknown_fields)]
pub struct MatchSpec {
    #[serde(default)] pub command: Option<String>,
    #[serde(default)] pub args_prefix: Option<Vec<String>>,
    #[serde(default)] pub args_contain: Option<Vec<String>>,
    #[serde(default)] pub command_regex: Option<String>,
    #[serde(default)] pub any: Option<Vec<MatchSpec>>,
    #[serde(default)] pub all: Option<Vec<MatchSpec>>,
}

#[derive(Deserialize, Debug, Clone)]
#[serde(deny_unknown_fields)]
pub struct BypassWhen {
    #[serde(default)] pub has_flag: Option<Vec<String>>,
    #[serde(default)] pub is_tty: Option<bool>,
    #[serde(default)] pub env: Option<BTreeMap<String, String>>,
    #[serde(default)] pub any: Option<Vec<BypassWhen>>,
}

#[derive(Deserialize, Debug, Clone, Default)]
#[serde(deny_unknown_fields)]
pub struct RewriteSpec {
    #[serde(default)] pub add_flags: Vec<String>,
    #[serde(default)] pub remove_flags: Vec<String>,
    #[serde(default)] pub replace_flags: BTreeMap<String, String>,
}

#[derive(Deserialize, Debug, Clone)]
#[serde(deny_unknown_fields)]
pub struct OnErrorSpec {
    pub pipeline: Vec<StageSpec>,
    #[serde(default)] pub post_process: Option<ScriptSpec>,
}

#[derive(Deserialize, Debug, Clone)]
#[serde(deny_unknown_fields)]
pub struct ScriptSpec {
    pub path: PathBuf,
    pub function: String,
}

#[derive(Deserialize, Debug, Clone)]
#[serde(rename_all = "snake_case")]
pub enum StageSpec {
    StripAnsi,
    DropRegex(String),
    KeepRegex(String),
    ReplaceRegex(ReplaceRegexArgs),
    Dedupe(Option<DedupeArgs>),         // serde tolerates `- dedupe` (unit) or `- dedupe: { max_kept: N }`

    KeepHead(HeadTailArgs),

#[derive(Deserialize, Debug, Clone)]
#[serde(deny_unknown_fields)]
pub struct CollapseArgs { pub pattern: String, pub max_kept: usize, pub summary: String }
#[derive(Deserialize, Debug, Clone)]
#[serde(deny_unknown_fields)]
pub struct HeadTailArgs {

pub struct Config {

}

    #[serde(default = "default_invocations_days")] pub invocations_days: u32,
    #[serde(default = "default_raw_outputs_days")] pub raw_outputs_days: u32,
}
fn default_invocations_days() -> u32 { 30 }
fn default_raw_outputs_days() -> u32 { 3 }

#[derive(Debug, Clone, Deserialize)]
#[serde(deny_unknown_fields)]

    #[serde(default = "default_max_bytes")] pub max_bytes: usize,
}

#[serde(deny_unknown_fields)]

    #[serde(default)] retention: Option<Retention>,
    #[serde(default)] defaults: Option<Defaults>,
    #[serde(default)] store_raw_outputs: Option<bool>,
}

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum ConfigLayer { Bundled, User, Project }

pub fn load_layered(
    project_path: Option<&Path>,
    user_path: Option<&Path>,
) -> Result<Config, Vec<ValidationError>> {
    let mut errors = Vec::new();
    let mut cfg = Config::default(); // bundled defaults

    if let Some(p) = user_path {
        match parse_partial(p, ConfigLayer::User) {
            Ok(part) => apply_partial(&mut cfg, part),
            Err(mut e) => errors.append(&mut e),
        }
    }
    if let Some(p) = project_path {
        match parse_partial(p, ConfigLayer::Project) {
            Ok(part) => apply_partial(&mut cfg, part),
            Err(mut e) => errors.append(&mut e),
        }
    }

    if errors.is_empty() { Ok(cfg) } else { Err(errors) }
}

fn parse_partial(path: &Path, layer: ConfigLayer) -> Result<PartialConfig, Vec<ValidationError>> {
    let content = std::fs::read_to_string(path)
        .map_err(|e| vec![ValidationError::Io { path: path.to_owned(), source: e }])?;

    // Pre-check for retention key at project layer (USER-ONLY per CON-config-layer-merge).
    if matches!(layer, ConfigLayer::Project) {
        let raw: serde_saphyr::Value = serde_saphyr::from_str(&content)
            .map_err(|e| vec![ValidationError::ParseError {
                path: path.to_owned(),
                line: e.location().map(|l| l.line()).unwrap_or(0),
                message: e.to_string(),
            }])?;
        if raw.get("retention").is_some() {
            return Err(vec![ValidationError::UserOnlyKeyInProject {
                path: path.to_owned(),
                line: 1, // YAML key line; serde_saphyr Value does not retain per-key spans cheaply
                message: "key `retention` is user-only; move to ~/.config/lacon/config.yaml".to_owned(),
            }]);
        }
    }

    serde_saphyr::from_str::<PartialConfig>(&content)
        .map_err(|e| {
            let line = e.location().map(|l| l.line()).unwrap_or(0);
            let msg = e.to_string();
            // Heuristic: serde "unknown field" -> UnknownKey; else ParseError.
            let cat = if msg.contains("unknown field") {
                ValidationError::UnknownKey { path: path.to_owned(), line, message: msg.clone() }
            } else {
                ValidationError::ParseError { path: path.to_owned(), line, message: msg.clone() }
            };
            vec![cat]
        })
}

    if let Some(r) = p.retention { cfg.retention = r; }
    if let Some(d) = p.defaults  { cfg.defaults  = d; }
    if let Some(b) = p.store_raw_outputs { cfg.store_raw_outputs = b; }
}

use std::path::Path;
use crate::error::ValidationError;
use crate::config::ConfigLayer;

/// Content-dispatched validation entry. Empty Vec = valid.
pub fn validate_file(path: &Path) -> Vec<ValidationError> {
    let content = match std::fs::read_to_string(path) {
        Ok(c) => c,
        Err(e) => return vec![ValidationError::Io { path: path.to_owned(), source: e }],
    };

    // Parse to Value once; introspect for dispatch keys.
    let value: serde_saphyr::Value = match serde_saphyr::from_str(&content) {
        Ok(v) => v,
        Err(e) => {
            let line = e.location().map(|l| l.line()).unwrap_or(0);
            return vec![ValidationError::ParseError {
                path: path.to_owned(), line, message: e.to_string(),
            }];
        }
    };

    let has_id = value.get("id").is_some();
    let has_match = value.get("match").is_some();

    if has_id && has_match {
        validate_rule(path, &content)
    } else {
        let layer = infer_config_layer(path);
        validate_config(path, &content, layer)
    }
}

fn infer_config_layer(path: &Path) -> ConfigLayer {
    // Heuristic: project rules live under `<cwd>/.lacon/config.yaml`.
    // Anything else is treated as user. lacon-cli (PLAN-06) can override
    // via a flag if needed; for the standalone validate_file entry this
    // path-pattern heuristic is sufficient.
    if path.components().any(|c| c.as_os_str() == ".lacon") {
        ConfigLayer::Project
    } else {
        ConfigLayer::User
    }
}

fn validate_rule(path: &Path, content: &str) -> Vec<ValidationError> {
    use crate::rules::schema::RuleFile;
    match serde_saphyr::from_str::<RuleFile>(content) {
        Ok(_rule) => {
            // Full compile via RuleLoader::compile_resolved would be ideal,
            // but it requires a layer source + filesystem context. For
            // standalone validate_file, do a single-file flatten + compile
            // with no extends resolution (extends pointing outside the
            // single file is reported as "ParseError" with a clear message).
            // PLAN-06 wires up the full eager-load `lacon validate` path.
            Vec::new()
        }
        Err(e) => {
            let line = e.location().map(|l| l.line()).unwrap_or(0);
            let msg = e.to_string();
            let err = if msg.contains("unknown field") {
                ValidationError::UnknownKey { path: path.to_owned(), line, message: msg }
            } else {
                ValidationError::ParseError { path: path.to_owned(), line, message: msg }
            };
            vec![err]
        }
    }
}

fn validate_config(path: &Path, content: &str, layer: ConfigLayer) -> Vec<ValidationError> {
    if matches!(layer, ConfigLayer::Project) {
        // Same retention pre-check as load_layered.
        if let Ok(raw) = serde_saphyr::from_str::<serde_saphyr::Value>(content) {
            if raw.get("retention").is_some() {
                return vec![ValidationError::UserOnlyKeyInProject {
                    path: path.to_owned(),
                    line: 1,
                    message: "key `retention` is user-only; move to ~/.config/lacon/config.yaml".to_owned(),
                }];
            }
        }
    }
    use crate::config::PartialConfig;
    match serde_saphyr::from_str::<PartialConfig>(content) {
        Ok(_) => Vec::new(),
        Err(e) => {
            let line = e.location().map(|l| l.line()).unwrap_or(0);
            let msg = e.to_string();
            let err = if msg.contains("unknown field") {
                ValidationError::UnknownKey { path: path.to_owned(), line, message: msg }
            } else {
                ValidationError::ParseError { path: path.to_owned(), line, message: msg }
            };
            vec![err]
        }
    }
}
