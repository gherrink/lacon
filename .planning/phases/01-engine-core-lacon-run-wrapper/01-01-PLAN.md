Create `crates/lacon-core/tests/wave0_smoke.rs` with two smoke tests that resolve RESEARCH.md open questions 1 and 2:

```rust
//! Wave 0 smoke tests for Phase 1.
//!
//! These tests exist to settle two open questions from RESEARCH.md before
//! PLAN-03 commits to the loader design:
//!
//! 1. Does `serde-saphyr` 0.0.26 expose a `Value` type adequate for the
//!    `lacon validate` content-dispatch check (top-level `id` AND `match`)?
//! 2. Does `starlark` 0.13 compile under the workspace MSRV of 1.80?
//!
//! Failure modes:
//! - If serde-saphyr lacks Value support: PLAN-03 will need to fall back to
//!   the underlying `saphyr` parser crate and hand-roll key inspection.
//!   Document the fallback in PLAN-03 acceptance.
//! - If starlark requires a higher MSRV: bump workspace `rust-version` in
//!   Cargo.toml accordingly. The active toolchain (1.94.1) almost certainly
//!   satisfies any modern crate's MSRV; this is a tripwire only.

#[test]
fn smoke_serde_saphyr_value_dispatch() {
    // Mimics the D-17 `lacon validate` dispatch: parse YAML to a generic
    // value type, look for top-level `id` AND `match`. If both present →
    // rule file. Otherwise → config file.

    let rule_yaml = r#"
id: example
match:
  command: echo
pipeline:
  - strip_ansi
"#;
    let config_yaml = r#"
defaults:
  max_bytes: 16384
"#;

    // serde-saphyr 0.0.26 deserialises into `serde_saphyr::Value` (mirrors
    // serde_yaml::Value). If this trait bound fails to resolve, the test
    // will not compile and PLAN-03 must use the fallback path.
    let rule_val: serde_saphyr::Value = serde_saphyr::from_str(rule_yaml)
        .expect("rule yaml parses");
    let config_val: serde_saphyr::Value = serde_saphyr::from_str(config_yaml)
        .expect("config yaml parses");

    // Assert dispatch keys are introspectable. Exact API surface
    // (`get`, `as_mapping`, indexing) is what PLAN-03 will use; if it
    // differs we want the failure here, not deep in PLAN-03.
    assert!(
        rule_val.get("id").is_some(),
        "rule YAML must expose top-level `id` via Value::get"
    );
```
